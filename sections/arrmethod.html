Add & Remove Methods
heroes = ["Batman", "Flash", "Aquaman"]

# ── Adding ────────────────────────────────
heroes.append("Superman")    # adds to END
heroes.insert(0, "Wonder")  # inserts at index 0
heroes.extend(["Hulk", "Thor"]) # merges another list in

# ── Removing ──────────────────────────────
heroes.remove("Flash")     # removes by VALUE (first match)
last = heroes.pop()        # removes + returns last item
first = heroes.pop(0)      # removes + returns at index 0
heroes.clear()              # nukes the entire list → []

# ⚠️ remove() crashes if value not found!
# Safe pattern: check first
if "Flash" in heroes:
    heroes.remove("Flash")
Sorting & Ordering
nums = [3, 1, 4, 1, 5, 9]
names = ["Zara", "Alice", "Bob"]

# .sort() — modifies the list IN PLACE
nums.sort()                # [1, 1, 3, 4, 5, 9]
nums.sort(reverse=True)   # [9, 5, 4, 3, 1, 1]

# sorted() — returns a NEW list (original untouched)
sorted_names = sorted(names)          # ['Alice','Bob','Zara']
by_length = sorted(names, key=len)   # sort by name length

# reverse() — flips in place
nums.reverse()    # reverses current order

# 📌 Rule of thumb:
# .sort()    = mutates list, returns None
# sorted()   = safe, returns new list
Searching & Info Methods
scores = [85, 92, 78, 92, 60]

# ── Finding ───────────────────────────────
scores.index(92)     # 1 (FIRST position of 92)
scores.count(92)     # 2 (appears twice)

# ── Global functions on lists ─────────────
len(scores)         # 5 (total items)
min(scores)         # 60
max(scores)         # 92
sum(scores)         # 407

# ── Membership check (fast!) ──────────────
92 in scores        # True
100 not in scores  # True

# ── Copy (avoid aliasing bugs!) ───────────
original = [1, 2, 3]
alias = original          # ⚠️ same object!
safe_copy = original.copy() # ✅ independent
Comprehensions & Power Patterns
# List Comprehension = create list in one line
# Syntax: [expression for item in iterable if condition]

nums = [1, 2, 3, 4, 5]

# Square every number
squares = [n**2 for n in nums]       # [1,4,9,16,25]

# Only even numbers
evens = [n for n in nums if n % 2 == 0] # [2,4]

# Flatten nested list
matrix = [[1,2],[3,4],[5,6]]
flat = [x for row in matrix for x in row] # [1,2,3,4,5,6]

# zip() — iterate two lists together
names = ["Alice", "Bob"]
ages  = [30, 25]
for name, age in zip(names, ages):
    print(f"{name} is {age}")
