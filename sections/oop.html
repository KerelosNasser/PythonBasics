Class Blueprint & The "self" Keyword
Creating Functions (`def`)

Functions let you reuse code blocks without copy-pasting.

# Define with 'def', return sends data back
def calculate_area(width, height):
    area = width * height
    return area

# Calling the function
result1 = calculate_area(5, 10)  # 50
result2 = calculate_area(7, 3)   # 21

# Default parameters (if user doesn't provide one)
def greet(name="Guest"):
    print(f"Hello {name}!")

greet()          # "Hello Guest!"
greet("Kero")    # "Hello Kero!"
OOP: Classes & Instances
# 1. Define the blueprint
class Player:
    # __init__ is the constructor (runs when created)
    def __init__(self, username):
        self.name = username
        self.level = 1  # Default value
    
    # A method (a function inside a class)
    def level_up(self):
        self.level += 1
        print(f"{self.name} reached level {self.level}!")

# 2. Create Instances (Objects)
p1 = Player("ShadowNinja")
p2 = Player("MageLord")

p1.level_up()  # ShadowNinja reached level 2!
