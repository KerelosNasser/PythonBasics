Case & Cleaning Methods
msg = "  hello world!  "

# ── Case transformations ──────────────────
msg.upper()       # "  HELLO WORLD!  "
msg.lower()       # "  hello world!  " (already lower)
msg.title()       # "  Hello World!  " (each word caps)
msg.capitalize()  # "  hello world!  " (only 1st char)
"hElLo".swapcase() # "HeLlO"

# ── Stripping whitespace ──────────────────
msg.strip()       # "hello world!" (both sides)
msg.lstrip()      # "hello world!  " (left only)
msg.rstrip()      # "  hello world!" (right only)

# Strip specific chars (not just spaces)
"###data###".strip("#") # "data"
Searching & Replacing
sentence = "Python is great, Python is fun!"

# ── Finding ───────────────────────────────
sentence.find("Python")    # 0  (first occurrence index)
sentence.find("Java")      # -1 (not found → safe!)
sentence.index("Java")     # ValueError (use find to be safe)

sentence.count("Python")   # 2 (appears twice)
sentence.startswith("Py") # True
sentence.endswith("fun!") # True

# ── Replacing ─────────────────────────────
sentence.replace("Python", "Kero")     # all occurrences
sentence.replace("Python", "Kero", 1) # first 1 only
Split, Join & Validation
# ── Splitting ─────────────────────────────
csv = "apple,banana,mango"
fruits = csv.split(",")       # ['apple', 'banana', 'mango']
words = "a b c".split()        # split on whitespace by default

# ── Joining ───────────────────────────────
", ".join(fruits)             # "apple, banana, mango"
" | ".join(["a","b","c"])   # "a | b | c"

# ── Checking content ──────────────────────
"12345".isdigit()    # True  (all digits)
"hello".isalpha()    # True  (all letters)
"abc123".isalnum()  # True  (letters + digits)
"   ".isspace()     # True  (only whitespace)
"HELLO".isupper()   # True
Formatting & Alignment
# ── Padding ────────────────────────────────
"007".zfill(6)           # "000007" (pad zeros left)
"hi".center(10)           # "    hi    " (center)
"hi".ljust(10)            # "hi        " (left align)
"hi".rjust(10)            # "        hi" (right align)
"hi".center(10, "*")      # "****hi****" (custom fill)

# ── f-String power moves ──────────────────
pi = 3.14159
print(f"Pi = {pi:.2f}")       # "Pi = 3.14" (2 decimals)
print(f"{'Name':<10} {'Score':>6}")   # aligned columns
print(f"{1000000:,}")         # "1,000,000" (comma sep)
print(f"{0.75:.0%}")          # "75%" (percentage)
