Combine multiple conditions together.

Match / Switch-Case (Python 3.10+)
The match Statement — Python's Switch-Case
Basic match — Replacing if/elif Chains
# Old way (messy if/elif chain)
if day == "Mon": print("Start of week")
elif day == "Fri": print("Weekend!")
else: print("Midweek")

# New way — match statement (Python 3.10+)
match day:
    case "Mon":
        print("Start of week")
    case "Fri" | "Sat" | "Sun":  # | = OR
        print("Weekend vibes!")
    case _:             # _ = default (catch-all)
        print("Midweek grind")
Capture Patterns & Guards
# Capture pattern: stores matched value in variable
score = 85
match score:
    case n if n >= 90:  # Guard: extra condition after 'if'
        print(f"A grade: {n}")
    case n if n >= 80:
        print(f"B grade: {n}")  # Prints this!
    case _:
        print("Keep studying")

# Matching sequences (lists/tuples)
point = (0, 5)
match point:
    case (0, 0): print("Origin")
    case (0, y): print(f"On Y-axis at y={y}")
    case (x, 0): print(f"On X-axis at x={x}")
    case (x, y): print(f"Point at {x},{y}")
Matching Dictionaries & Classes
# Match against dict shape (great for API responses!)
event = {"type": "click", "button": "left"}

match event:
    case {"type": "click", "button": btn}:
        print(f"{btn} click!")  # "left click!"
    case {"type": "keypress", "key": k}:
        print(f"Key pressed: {k}")
    case _:
        print("Unknown event")

# match vs if/elif: use match when comparing
# the SAME variable against multiple exact values
Real-World CLI Command Router
# Classic use-case: routing commands in a game/CLI
def handle_command(cmd):
    match cmd.split():         # Split "go north" → ["go","north"]
        case ["quit"]:
            print("Goodbye!")
        case ["go", direction]:
            print(f"Moving {direction}")
        case ["pick", "up", item]:
            print(f"Picked up {item}")
        case [unknown, *rest]:   # *rest = remaining words
            print(f"Unknown: {unknown}")

handle_command("go north")    # Moving north
handle_command("pick up sword") # Picked up sword

Sorting & Ordering
nums = [3, 1, 4, 1, 5, 9]
names = ["Zara", "Alice", "Bob"]

# .sort() — modifies the list IN PLACE
nums.sort()                # [1, 1, 3, 4, 5, 9]
nums.sort(reverse=True)   # [9, 5, 4, 3, 1, 1]

# sorted() — returns a NEW list (original untouched)
sorted_names = sorted(names)          # ['Alice','Bob','Zara']
by_length = sorted(names, key=len)   # sort by name length

# reverse() — flips in place
nums.reverse()    # reverses current order

# 📌 Rule of thumb:
# .sort()    = mutates list, returns None
# sorted()   = safe, returns new list

Comprehensions & Power Patterns
# List Comprehension = create list in one line
# Syntax: [expression for item in iterable if condition]

nums = [1, 2, 3, 4, 5]

# Square every number

Tkinter — Python's Built-In GUI Framework
Tkinter Architecture & Widget Tree
Your First Tkinter Window
import tkinter as tk

# 1. Create the root window (the app itself)
root = tk.Tk()
root.title("My First App")
root.geometry("400x300")    # width x height
root.resizable(False, False) # lock window size

# 2. Add widgets
label = tk.Label(root, text="Hello, Tkinter!",
                  font=("Arial", 18), fg="#22d3b0")
label.pack(pady=20)           # pady = vertical padding

btn = tk.Button(root, text="Click Me!",
                command=lambda: label.config(text="Clicked!"))
btn.pack()

# 3. Start the event loop (keep window open)
root.mainloop()  # blocks until window closes
Grid Layout & Entry Input
import tkinter as tk

root = tk.Tk()
root.title("Login Form")

# Labels in column 0, Entry boxes in column 1
tk.Label(root, text="Username:").grid(row=0, column=0, padx=8, pady=8)
entry_user = tk.Entry(root, width=20)
entry_user.grid(row=0, column=1)

tk.Label(root, text="Password:").grid(row=1, column=0, padx=8)
entry_pass = tk.Entry(root, show="*")  # hide with *
entry_pass.grid(row=1, column=1)

def login():
    user = entry_user.get()   # .get() reads entry value
    print(f"Logging in as: {user}")

tk.Button(root, text="Login", command=login).grid(row=2, column=1, pady=8)
root.mainloop()
StringVar, Checkbutton & Messagebox
import tkinter as tk
from tkinter import messagebox

root = tk.Tk()

# StringVar = reactive variable that updates the UI
name_var = tk.StringVar()
tk.Entry(root, textvariable=name_var).pack()

# IntVar for checkboxes (0 or 1)
agree_var = tk.IntVar()
tk.Checkbutton(root, text="I agree",
               variable=agree_var).pack()

def submit():
    if agree_var.get() == 0:
        messagebox.showwarning("Oops", "Please agree first!")
    else:
        messagebox.showinfo("Hi!", f"Hello {name_var.get()}!")

tk.Button(root, text="Submit", command=submit).pack()
root.mainloop()
Full Mini-App: Counter
import tkinter as tk

root = tk.Tk()
root.title("Counter App")
root.geometry("260x180")

count = tk.IntVar(value=0)

# Display label tied to the IntVar
tk.Label(root, textvariable=count,
         font=("Arial", 48, "bold"),
         fg="#22d3b0").pack(pady=16)

# Button row using frame
frame = tk.Frame(root)
frame.pack()
tk.Button(frame, text="−", font=("Arial", 20),
          command=lambda: count.set(count.get()-1)).pack(side="left", padx=8)
tk.Button(frame, text="+", font=("Arial", 20),
          command=lambda: count.set(count.get()+1)).pack(side="left", padx=8)
tk.Button(root, text="Reset",
          command=lambda: count.set(0)).pack(pady=8)
root.mainloop()
